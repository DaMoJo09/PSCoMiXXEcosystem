async function syncToStreamingPlatform(project, user) {
  const STREAMING_API = 'https://gamexclub.preview.emergentagent.com';
  const WEBHOOK_SECRET = 'whsec_45923106c2f0da38c1ec0d10e92accd6666bfeb3610c84de0144c56933bc3dc6';
  
  const bundle = {
    contract_version: 'v1',
    content_id: project.id,
    content_type: project.type || 'comic',
    title: project.name,
    description: project.description || '',
    cover_url: project.coverImage || project.cover_url,
    creator_ps_user_id: user.id,
    creator_display_name: user.displayName || user.name,
    visibility: 'public',
    tags: project.tags || [],

    pages: project.pages?.map((page, index) => ({
      page_number: index + 1,
      image_url: page.imageUrl || page.url
    })) || []
  };
  
  try {

      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-webhook-secret': WEBHOOK_SECRET
      },
      body: JSON.stringify(bundle)
    });
    
    const result = await response.json();
    console.log('‚úÖ Synced to streaming platform:', result);
    return result;
  } catch (error) {
